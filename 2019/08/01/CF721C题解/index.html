    题目简述

    一座城市里有 $n$ 个景点,景点从 $1$ 到 $n$ 进行编号有些景点之间设有单向的道路,Irina经过 $i$ 道路要时间 $t_i$ ,这 $n$ 个城市和这些道路形成无环图,Irina从 $1$ 号景点出发到 $n$ 号景点Irina想知道在行驶总时间不超过 $T$ 的情况下最多能游览几个景点(前提是他要到达 $n$ 号景点),以及他游览最多景点时依次经过的景点(包括 $1$ 和 $n$ )

    输入格式

    首行三个数 $n,m,T(2\leq n \leq 5000, 1 \leq m \leq 5000,1\leq T \leq 10^9)$ 随后 $m$ 行里每行三个数 $u_i,v_i,t_i(1\leq u_i,v_i \leq n,u_i \ne v_i,1\leq t_i \leq 10^9)$ 表示景点 $u_i$ 和景点 $v_i$ 之间有一条从 $u_i$ 到 $v_i$ 的有向道路,Irina需要花时间 $t_i$ 经过这条道路

    输出格式

    第一行输出 $k(2\leq k \leq n)$ ,数据保证有解. 第二行输出从 $1$ 到 $n$ 依次经过的景点编号，每两个景点编号之间有空格

    样例

    输入1

    4 3 13
    1 2 5
    2 3 7
    2 4 8

    输出1

    3
    1 2 4

    输入2

    6 6 7
    1 2 2
    1 3 3
    3 6 3
    2 4 2
    4 6 2
    6 5 1

    输出2

    4
    1 2 4 6

    输入3

    5 5 6
    1 3 3
    3 5 3
    1 2 2
    2 4 3
    4 5 2

    输出3

    3
    1 3 5

    因为是有向无环图，我们可以将该图的拓扑排序,

    $dp[i][j]$ 表示从 $ 1 $ 到 $i$ 的路程中经过 $j$ 个点(包括 $ 1 $ 和 $ i $ )的最短路径,通过拓扑序进行dp即可.

    $$ dp[i][j]=\min_{ver[k][i]=1}(dp[k][j-1]) $$

    设初值 $dp[1][1]=1$

    时间复杂度为 $O(nm)$

    代码:

    #include <bits/stdc++.h>
#define maxn 5010
#define inf 1000000001
#define rei register int
#define rel register ll
using namespace std;
struct node{
    int to,next,w;
}edge[maxn],n_edge[maxn];
int n,m,t;
int head[maxn],tot,n_tot,n_head[maxn];
int entry[maxn];
int dp[maxn][maxn],prevv[maxn][maxn];
void add(int u,int v,int w){
    edge[++tot].to=v,edge[tot].next=head[u],edge[tot].w=w,head[u]=tot; 
    n_edge[++n_tot].to=u,n_edge[n_tot].next=n_head[v],n_edge[n_tot].w=w,n_head[v]=n_tot;
}
int main(){
    scanf("%d%d%d",&n,&m,&t);
    for(rei i=1;i<=m;++i){
        int u,v,w;
        scanf("%d%d%d",&u,&v,&w);
        add(u,v,w);
        ++entry[v];
    }
    for(rei i=0;i<=n;++i)
     for(rei j=0;j<=n;++j)
      dp[i][j]=inf;
    int q[maxn],hhead=1,tail=0;
    for(rei i=1;i<=n;++i)
    if(!entry[i])
    q[++tail]=i;
    dp[1][1]=0;
    while(hhead<=tail){
        int p=q[hhead++];
        for(rei j=1;j<=n;++j)
        for(rei i=n_head[p];i;i=n_edge[i].next)
        if(dp[n_edge[i].to][j-1]+n_edge[i].w<dp[p][j])
        dp[p][j]=dp[n_edge[i].to][j-1]+n_edge[i].w,prevv[p][j]=n_edge[i].to;
        for(rei i=head[p];i;i=edge[i].next){
        entry[edge[i].to]--;
        if(!entry[edge[i].to])
        q[++tail]=edge[i].to;
        }
       }
    for(rei i=n;i;i--)
    if(dp[n][i]<=t){
        printf("%d\n",i);
        int tot=0;
        int s[maxn],top=0;
        for(rei j=n;j;j=prevv[j][i-tot],tot++)
        s[++top]=j;
        for(rei j=top;j;j--)
        printf("%d ",s[j]);
        return 0;
    }
    return 0;
}
