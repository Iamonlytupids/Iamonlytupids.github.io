    题目描述

    输入格式

    输入仅有一行,包含两个正整数 $q,n$ ,分别表示问题编号以及叶结点的个数。

    输出格式

    输出仅有一行，包含一个实数 $d$ ，四舍五入精确到小数点后 6 位。如果 $q = 1$ ，则 $d$ 表示叶结点平均深度的数学期望值；如果 $q = 2$ ，则 $d$ 表示树深度的数学期望值。

    样例

    输入1

    1 4

    输出1

    2.166667

    输入2

    2 4

    输出2

    2.666667

    输入3

    1 12

    输出3

    4.206421

    输入4

    2 12

    输出4

    5.916614

    数据范围

    第一眼看到期望就会想到DP

    我们尝试设计一个能够转移的DP状态

    先思考1问题

    我们可以考虑将操作的次数设计为DP的第一维

    这样我们就可以得到一个DP状态 $E[i]$ 表示经过 $i$ 次操作后叶节点的深度和的期望(问题1要求的是平均深度,最后将求得的期望除以最后的叶节点个数就可求出)

    那么我们如何转移这样一个状态呢

    在求 $E[i]$ 的时候,我们可以先将起始节点A进行一步操作

    然后我们可以将得到的两个叶子节点看作两个起始节点B,C,这样两个起始节点的操作数 $t$ 就会小于 $i$ 了,我们就可以尝试通过这两个节点和之前的dp状态来进行转移

    DP方程很好想
    这里两个节点操作后的叶子节点的深度和 $X,Y$ 相加后再加上叶子节点总数 $T$ 就是整棵树叶子结点的深度和Z,所以 $E(Z)=E(X+Y)+T=E(X)+E(Y)+T$
    $$E[i]=\Sigma^{i-1} _ {j=0} (E[j] + E[i-1-j]) * g[j][i-1-j] + i + 1$$
    初值: $E[0]=0$
    这里 $g[i][j]$ 初始状态表示操作一次以后，在A的左子树中操作了 $i$ 次，在A的右子树中操作了 $j$ 次的概率

    下面考虑如何得到 $g[i][j]$

    我们可以考虑记录 $g_e[i][j]$, 它表示初始状态操作一次以后，在A的左子树中操作了 $i$ 次，在A的右子树中操作了 $j$ 次,其中最后一次操作在A的左子树中发生的概率

    两者的转移方程如下

    $$g[i][j]=\Sigma^j_{k=0}g_e[i][k]*\Pi^{j-1} _ {p=k}(p+1)/(i+p+2)$$

    $$g_e[i][j]=g[i-1][j]*i/(i+j+1)$$

    初值: $g[0][0]=g_e[0][0]=1$
    这样问题1就解决了

    下面考虑问题2
    问题2的处理方法大同小异
    直接用期望DP不好DP
    我们可以通过概率进行DP

    设 $P[i][j]$ 表示在初始状态操作 $i$ 次后最大树深是 $j$ 的概率

    $$P[i][j]=\Sigma^{i-1} _ {k=0}\Sigma_{\max(x,y)+1=j}P[k][x] * P[i-1-k][y] * g[k][i-1-k]$$

    初值: $P[0][0]=1$

    最后再计算期望即可

    时间复杂度为 $O(n^3)$

    代码:
    #include <bits/stdc++.h>
#define rei register int
#define rel register ll
#define maxn 110
typedef long long ll;
typedef double db;
typedef long double ldb;
typedef unsigned long long ull;
using namespace std;
int q,n;
ldb g[maxn][maxn][2]; // g[i][j][0] g[i][j] g[i][j][1] 表示g_e[i][j]
ldb E[maxn];
ldb P[maxn][maxn];
void Pre_DP(){
    g[0][0][0] = g[0][0][1] = 1;
    for(rei tot = 1 ; tot <= n - 2; ++ tot)
     for(rei i = 0 ; i <= tot ; ++ i){
      if( i - 1 >= 0 )
      g[i][tot - i][1] = g[i - 1][tot - i][0] * (ldb) i / (ldb) (tot + 1);
      ldb k = 1; 
      for(rei j = tot - i ; j >= 0 ; --j)
      g[i][tot - i][0] += g[i][j][1] * k , k *= (ldb) j / (ldb) (i + j + 1);
     }
}
int main(){
    ios::sync_with_stdio(false);
    cin >> q >> n ;
    Pre_DP();
    if( q == 1 ){
        for(rei i = 1 ; i <= n - 1; ++i ){
         for(rei j = 0; j <= i - 1 ; ++j )
          E[i] += ( E[j] + E[i - j - 1] ) * g[j][i - j - 1][0];
          E[i] += (ldb) i + (ldb) 1;
        }
        printf("%.6Lf\n",E[n - 1] / n );
    }
    if( q == 2 ){
        P[0][0]=1;
        for(rei i = 1 ; i <= n - 1 ; ++i )
         for(rei j = 0 ; j <= i - 1 ; ++j ){
             ldb cnt = 0;
             for(rei k = 1 ; k <= j ; ++k )
             cnt += P[i - 1 - j][k - 1] , P[i][k + 1] += P[j][k] * cnt * g[j][i - 1 - j][0];
             cnt = 0;
             for(rei k = 0 ; k <= j ; ++k )
             cnt += P[i - 1 -j][k], P[i][k + 1] += P[j][k] * cnt * g[j][i - 1 - j][0];
         }
        ldb ans = 0;
        for(rei i = 1 ; i <= n - 1 ; ++i )
        ans += (ldb) i * P[n - 1][i];
        printf("%.6Lf",ans);
    }
    return 0;
}
